# 检查文件是否存在
# 存在结果0，否则1
# 检查文件是否存在并输出
test -e /abc && echo "exist" || echo "not exist"

# 存在
+----+-------------------------------------+
| -e | 文件是否存在
+----+-------------------------------------+
| -f | 文件是否存在，且为 file
+----+-------------------------------------+
| -d | 文件是否存在，且为 directory
+----+-------------------------------------+
| -b | 文件是否存在，且为 block device
+----+-------------------------------------+
| -c | 文件是否存在，且为 character device
+----+-------------------------------------+
| -S | 文件是否存在，且为 socket file
+----+-------------------------------------+
| -p | 文件是否存在，且为 pipe file
+----+-------------------------------------+
| -L | 文件是否存在，且为 link file
+----+-------------------------------------+

# 权限
+----+--------------------------------------+
| -r | 文件是否存在，且具备 r 权限
+----+--------------------------------------+
| -w | 文件是否存在，且具备 w 权限
+----+--------------------------------------+
| -x | 文件是否存在，且具备 x 权限
+----+--------------------------------------+
| -u | 文件是否存在，且具有 SUID 属性
+----+--------------------------------------+
| -g | 文件是否存在，且具有 SGID 属性‘
+----+--------------------------------------+
| -k | 文件是否存在，且具有 sticky bit 属性
+----+--------------------------------------+
| -s | 文件是否存在，且非空
+----+--------------------------------------+

# 文件比较
# test file1 -nt file2
+-----+---------------------------------------------------+
| -nt | 文件1比文件2新，new than
+-----+---------------------------------------------------+
| -ot | 文件1比文件2旧，old than
+-----+---------------------------------------------------+
| -ef | 两个文件是否为统一文件，判断依据是指向同一个inode
+-----+---------------------------------------------------+

# 整数比较
# test n -eq m
| -eq | n == m，equal
+-----+-------------------------------+
| -ne | n != m，not equal
+-----+-------------------------------+
| -gt | n > m，greater than
+-----+-------------------------------+
| -lt | n < m，less than
+-----+-------------------------------+
| -ge | n >= m，greater than or equal
+-----+-------------------------------+
| -le | n <= m，less than or equal
+-----+-------------------------------+

# 字符串
+-------------------------+----------------+
| test -z string          | 判断字符串为空
+-------------------------+----------------+
| test -n string          | 判断字符串非空
+-------------------------+----------------+
| test string1 == string2 | 判断字符串相等
+-------------------------+----------------+
| test string1 != string2 | 判断字符串不等
+-------------------------+----------------+

# 多重
+----+-----+-------------------------+
| -a | and | test -f file -a -x file
+----+-----+-------------------------+
| -o | or  | test -r file -o -w file
+----+-----+-------------------------+
| !  | 非  | test ! -x file
+----+-----+-------------------------+
